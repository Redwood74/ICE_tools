    CMD -->|check-once| CFG["Load Config

    CMD -->|check-batch| BCFG["Load Config + People YAML

    CMD -->|smoke-test| SMOKE["Classify Fixtures

    VALIDATE -->|No| ERR_CFG["❌ ConfigError

    VALIDATE -->|Yes| RUN["execute_run()

    BCFG --> BVAL{"Config valid?"}
    BVAL -->|Yes| BATCH["execute_batch()

execute_run() + delay"]
    BVAL -->|No| ERR_CFG

    BATCH --> RUN

    RUN --> RUNID["Generate run_id

    RETRY --> BEST["best_state_from_run()

    BEST --> BRANCH{Best state?}

    BRANCH -->|LIKELY_POSITIVE| DEDUP{"New result?

    DEDUP -->|Already seen| SKIP["Log: duplicate,

    BRANCH -->|AMBIGUOUS_REVIEW| AMB["⚠️ Log warning

    AMB --> PERSIST

2. Browser Automation: run_with_retries → run_single_attempt

flowchart TD
    RWR["run_with_retries()
a_number, country, attempts=4"]
    RWR --> RESET["reset_selector_health()"]
    RESET --> PW["Start Playwright
sync_playwright()"]
    PW --> LOOP["FOR attempt 1..N"]

    LOOP --> DELAY{"attempt > 1?"}
    DELAY -->|Yes| WAIT["Sleep
delay + random jitter"]
    DELAY -->|No| ATTEMPT
    WAIT --> ATTEMPT

    ATTEMPT["run_single_attempt()"]

    ATTEMPT --> LAUNCH["Launch Chromium
--disable-blink-features=AutomationControlled
--no-sandbox, --disable-infobars"]
    LAUNCH --> CTX["Create browser context
viewport 1280×900
Chrome user-agent"]
    CTX --> PAGE["new_page()"]
    PAGE --> STEALTH["Stealth().apply_stealth_sync(page)
Patch webdriver, plugins, chrome.runtime,
WebGL, languages, permissions"]

    STEALTH --> NAV["page.goto(locator.ice.gov)
wait_until=domcontentloaded"]
    NAV --> SETTLE["wait_for_load_state('networkidle')
+ random 1.5–3s wait
+ random mouse movement"]

    SETTLE --> FILL_A["Fill A-Number
resolve_locator(A_NUMBER_INPUT)
click → press_sequentially(digits)
40-90ms per key → press Tab"]
    FILL_A --> FILL_C["Select Country
resolve_locator(COUNTRY_SELECT)
select_option(value/label)"]
    FILL_C --> CLICK_S["Click Search
resolve_locator(SEARCH_BUTTON)
button.click()"]

    CLICK_S --> WAIT_R["_wait_for_result()
Wait for #resultsPage, #detentionPage,
'search results:', 'no records',
'internal error', 'our apologies'
timeout: 20s"]

    WAIT_R --> EXTRACT["Extract page text
Result container → body fallback"]
    EXTRACT --> CLASSIFY["classify_page_text()
→ ResultState"]
    CLASSIFY --> SAVE_ART["Save artifacts
screenshot + HTML + text"]

    SAVE_ART --> FAC{"Detention
facility found?"}
    FAC -->|Yes| DETAIL["Click facility link
Extract detail page
(address, ERO office, phone)"]
    DETAIL --> MORE{"More info
URL available?"}
    MORE -->|Yes| TABS["Open facility info page
Click each tab
Extract text, phones,
emails, links"]
    MORE -->|No| CHECK_BOT
    TABS --> SAVE_D["Save detail + facility
info artifacts"]
    SAVE_D --> CHECK_BOT
    FAC -->|No| CHECK_BOT

    CHECK_BOT{"State ==
BOT_CHALLENGE?"}
    CHECK_BOT -->|Yes| BOT_ERR["🚫 raise BotChallengeError"]
    CHECK_BOT -->|No| RETURN["Return SearchResult"]

    RETURN --> EARLY{"State ==
LIKELY_POSITIVE?"}
    EARLY -->|Yes| STOP["Break loop early ✅"]
    EARLY -->|No| NEXT{"More attempts?"}
    NEXT -->|Yes| LOOP
    NEXT -->|No| HEALTH["log_selector_health()
Return all results"]

    BOT_ERR --> HEALTH
    STOP --> HEALTH

3. Page Classification Pipeline

flowchart TD
    INPUT["classify_page_text()
raw_text + page_title + http_status"]
    INPUT --> EMPTY{"Text empty?"}
    EMPTY -->|Yes| RET_ERR["Return ERROR"]

    EMPTY -->|No| NORM["Normalise: lowercase + strip
full_text = title + raw_text"]

    NORM --> BOT{"Bot phrases found?
captcha, verify you are human,
access denied, forbidden,
unusual traffic, rate limit...
OR http_status ∈ {403, 429, 503}"}
    BOT -->|Yes| RET_BOT["Return BOT_CHALLENGE_OR_BLOCKED"]

    BOT -->|No| ZERO{"Zero-result phrases?
0 search results, no records found,
not found, no match,
unable to locate"}
    ZERO -->|Yes| RET_ZERO["Return ZERO_RESULT"]

    ZERO -->|No| AMB{"Ambiguous/error phrases?
internal error, our apologies,
an error occurred,
go to locator home"}
    AMB -->|Yes| RET_AMB["Return AMBIGUOUS_REVIEW"]

    AMB -->|No| POS{"Positive phrases ≥ 2?
1 search result, search results:,
book-in date, detainee information,
current detention facility,
status: in ice custody"}
    POS -->|Yes| RET_POS["Return LIKELY_POSITIVE ✅"]

    POS -->|No| SHORT{"Text < 50 chars?"}
    SHORT -->|Yes| RET_ERR2["Return ERROR
(incomplete page load)"]
    SHORT -->|No| RET_AMB2["Return AMBIGUOUS_REVIEW
(content unclear)"]

4. Selector Resolution Strategy

flowchart TD
    RESOLVE["resolve_locator(page, SelectorGroup)"]
    RESOLVE --> LOOP["FOR idx, selector in candidates"]
    LOOP --> TRY["page.locator(selector)"]
    TRY --> COUNT{"locator.count() > 0?"}

    COUNT -->|Yes| HEUR{"idx ≥ heuristic_index?"}
    HEUR -->|No| OK["✅ Log DEBUG
Return locator.first"]
    HEUR -->|Yes| WARN["⚠️ Log WARNING
'resolved via HEURISTIC fallback'
Return locator.first"]

    COUNT -->|No| NEXT{"More candidates?"}
    NEXT -->|Yes| LOOP
    NEXT -->|No| FAIL["❌ Log WARNING
'No selector resolved'
Return None"]

    EX["Example: A_NUMBER_INPUT
11 candidates"]
    EX --> L1["[0-1] aria-label: Alien / A-Number"]
    L1 --> L2["[2-4] placeholder: Alien / A-Number"]
    L2 --> L3["[5] role=textbox"]
    L3 --> L4["[6-9] name/id attributes"]
    L4 --> L5["[10-11] 🔶 HEURISTIC:
first text input / visible input"]

    style FAIL fill:#b71c1c,color:#fff
